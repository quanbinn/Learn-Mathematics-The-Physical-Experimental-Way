### 这里使用住房的面积和市场上售卖价格的相对关系作为例子。

import matplotlib.pyplot as plt

x_of_area = [40,50,60,70,80,90,100,110,120,130] # 单位是平方米，40代表这个房子的面积是40平方米。

plt.scatter(x_of_area, y_of_price)
plt.show()
