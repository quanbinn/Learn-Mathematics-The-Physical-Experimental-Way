```python
import matplotlib.pyplot as plt
import numpy as np

x = np.linspace(0, 2.5, num=26)
print(x)
# y = np.sin(x)  
# y = np.power(np.e, np.sqrt(np.sin(x)))
# y = np.power(np.e, x)
y = np.sqrt(np.sin(x))
print(y)
plt.scatter(x, y)

plt.show()
```
